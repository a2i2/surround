import os
import json
import logging
from surround import Assembler
from .stages import {project_name}Data, Main
from .runners import WebRunner

logging.basicConfig(level=logging.INFO)

def main():
    data = {project_name}Data()
    assembler = Assembler("Default project", data, Main())

    WebRunner(assembler).run()

    logging.info(data.output_data)

if __name__ == "__main__":
    main()
